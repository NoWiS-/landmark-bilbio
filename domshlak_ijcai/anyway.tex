\begin{frame}
  \frametitle{LM-Cut}

  \begin{block}{}
    \begin{itemize}
      \item Landmark heuristic
      \item cost-partitioning additive $h^{max}$
    \end{itemize}
  \end{block}

  \begin{block}{Algo}
    \begin{algorithm*}[H]
      \footnotesize
      \caption{LM-Cut}
      \SetKwInOut{Input}{input}

      \Input{$(A, O, I, G)$ une tâche de planification $P$, $s$ l'état à évaluer}

      $h~←~h^{max}(s)$\;
      \lIf{$(h = 0)\vee(h = \infty)$}{ \Return $h$ }
      
      $h~←~0$\;
      \Repeat{$h^{max}(s) > 0$}{
        $L~←$ Ensemble des Actions LM calculé avec $P^+$ (avec coût non nul)

        $c~←~min_{l \in L} cost(l)$\;
        $h~←~h~+~c$\;
        \lFor{$l \in L$}{$cost(l)~←~cost(l) - c$\;}
      }
      \Return $h$\;


    \end{algorithm*}
  \end{block}
\end{frame}
